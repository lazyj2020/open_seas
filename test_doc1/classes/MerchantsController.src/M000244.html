  def show
    @merchant = Merchant.find(params[:id])
    if (current_ship.port_id != @merchant.port_id)
      flash[:notice] = 'You\'re not at that port'
      respond_to do |format|
        format.html { redirect_to(Port.find(@merchant.port_id)) }
        format.xml  { render :xml => @merchant }
      end
    else
      session[:merchant_id] = params[:id]
      respond_to do |format|
        format.html # show.html.erb
        format.xml  { render :xml => @merchant }
      end
    end
  end
