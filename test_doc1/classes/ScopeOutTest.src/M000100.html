  def using_public_scopes(klass)
    raise "must pass a block" unless block_given?
    $with_methods = (klass.methods-['with_scope']).grep(/^with_.*$/)
    class << klass
      public *$with_methods
    end
    yield
    class << klass
      protected *$with_methods
    end
  end
