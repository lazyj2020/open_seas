  def sell
    @ship_item = ShipItem.new
    session[:value] = params[:value].to_i
    session[:item] = params[:item].to_i
    respond_to do |format|
      format.html # sell.html.erb
      format.xml  { render :xml => @ship_item }
    end
  end
