 def forward
    @original = current_user.received_messages.find(params[:id])
    
    subject = @original.subject.sub(/^/, "Fwd: ")
    body = @original.body.gsub(/^/, "> ")
    body = body + "\n"
    @message = current_user.sent_messages.build(:subject => subject, :body => body)
    render :template => "sent/new"
  end
